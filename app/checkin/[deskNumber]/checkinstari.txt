// LRU Cache klasa za preloaded slike
class LRUCache {
  private cache: Map<string, { value: boolean; timestamp: number }>;
  private maxSize: number;
  private maxAgeMs: number;

  constructor(maxSize = 50, maxAgeMs = 4 * 60 * 60 * 1000) { // 4 sata
    this.cache = new Map();
    this.maxSize = maxSize;
    this.maxAgeMs = maxAgeMs;
  }

  has(key: string): boolean {
    const entry = this.cache.get(key);
    if (!entry) return false;
    // Ekspajraj stare unose
    if (Date.now() - entry.timestamp > this.maxAgeMs) {
      this.cache.delete(key);
      return false;
    }
    return true;
  }

  set(key: string, value: boolean): void {
    if (this.cache.size >= this.maxSize) {
      const firstKey = this.cache.keys().next().value;
      if (firstKey) this.cache.delete(firstKey);
    }
    if (this.cache.has(key)) this.cache.delete(key);
    this.cache.set(key, { value, timestamp: Date.now() });
  }

  get size(): number { return this.cache.size; }
  clear(): void { this.cache.clear(); }
}

// Preload funkcija za slike (radi samo na klijentskoj strani)
const preloadImage = (src: string): Promise<void> => {
  return new Promise((resolve) => {
    if (!src || typeof window === 'undefined') {
      resolve();
      return;
    }
    
    // Koristimo window.Image za klijentsku stranu
    const img = new window.Image();
    img.src = src;
    
    img.onload = () => resolve();
    img.onerror = () => {
      console.warn(`Failed to preload image: ${src}`);
      resolve(); // Ne odbij promise, samo loguj
    };
    
    // Timeout za slučaj da se slika nikad ne učita
    setTimeout(() => resolve(), 2000);
  });
};

// Helper funkcija za formatiranje preostalog vremena
const formatTimeRemaining = (minutes: number): string => {
  if (minutes >= 60) {
    const hours = Math.floor(minutes / 60);
    const remainingMinutes = minutes % 60;
    if (remainingMinutes === 0) {
      return `Check-in opens in ${hours} ${hours === 1 ? 'hour' : 'hours'}`;
    }
    return `Check-in opens in ${hours}h ${remainingMinutes}m`;
  }
  return `Check-in opens in ${minutes} ${minutes === 1 ? 'minute' : 'minutes'}`;
};

// Helper funkcija za formatiranje vremena zatvaranja
const formatClosingTime = (minutes: number): string => {
  if (minutes <= 0) return 'Check-in closed';
  if (minutes >= 60) {
    const hours = Math.floor(minutes / 60);
    const remainingMinutes = minutes % 60;
    if (remainingMinutes === 0) {
      return `Check-in closes in ${hours} ${hours === 1 ? 'hour' : 'hours'}`;
    }
    return `Check-in closes in ${hours}h ${remainingMinutes}m`;
  }
  return `Check-in closes in ${minutes} ${minutes === 1 ? 'minute' : 'minutes'}`;
};

    const allFlightsForDesk = data.departures.filter(f => {
      if (!f.CheckInDesk) return false;
      const deskNumbers = f.CheckInDesk.split(',').map((d: string) => d.trim());
      return (
        deskNumbers.includes(deskNumberParam) ||
        deskNumbers.includes(deskNumberParam.replace(/^0+/, '')) ||
        deskNumbers.includes(deskNumberParam.padStart(2, '0'))
      );
    });
 
    const flightsWithTime = allFlightsForDesk
      .map(flight => {
        let departureTime: Date | null = null;
        let isToday = true;
 
        if (flight.ScheduledDepartureTime) {
          if (flight.ScheduledDepartureTime.includes('T')) {
            departureTime = new Date(flight.ScheduledDepartureTime);
            isToday = departureTime.toDateString() === today;
          } else {
            const [hours, minutes] = flight.ScheduledDepartureTime.split(':').map(Number);
            departureTime = new Date(now);
            departureTime.setHours(hours, minutes, 0, 0);
            const minutesDiff = (now.getTime() - departureTime.getTime()) / (1000 * 60);
            if (minutesDiff > 30) isToday = false;
          }
        }
 
        return { ...flight, departureTime, isToday };
      })
      .filter(f => f.departureTime !== null) as (EnhancedFlight & {
        departureTime: Date;
        isToday: boolean;
      })[];
 
    const sortedFlights = flightsWithTime.sort(
      (a, b) => a.departureTime.getTime() - b.departureTime.getTime()
    );
 
    const futureTodayFlights = sortedFlights.filter(
      f =>
        f.isToday &&
        f.departureTime > now &&
        !f.StatusEN?.toLowerCase().includes('cancelled')
    );
 
    const currentFlight: EnhancedFlight | null =
      futureTodayFlights.length > 0 ? futureTodayFlights[0] : null;
 
    const updateTime = new Date().toLocaleTimeString('en-GB');
 
    if (!isMountedRef.current) return;
